\section{Exigences fonctionnelles et acteurs}

\textbf{Acteurs du système :}
\begin{enumerate}
    \item \textbf{Utilisateur Final} (Visiteur web/mobile)
    \item \textbf{Analyste Météo / Climatologue} (Utilisateur avancé)
    \item \textbf{Data Engineer / Administrateur Système}
    \item \textbf{Systèmes} (API Météo externe et Apache Airflow interne)
\end{enumerate}

\textbf{Exigences fonctionnelles :}
\begin{itemize}
    \item Collecter et ingérer les données météorologiques (Historique, Temps Réel, Prévisions).
    \item Consulter les conditions actuelles, l'historique et les prévisions sur un tableau de bord.
    \item Filtrer les données par gouvernorat (24 capitales régionales).
    \item Détecter les anomalies climatiques (pics de température/vent) et générer des alertes.
    \item Calculer des statistiques avancées (moyennes, corrélations).
    \item Superviser l'infrastructure (Kafka, DAGs Airflow) et la qualité des données.
\end{itemize}

\section{Exigences non fonctionnelles}
\begin{itemize}
    \item \textbf{Performance :} Actualisation fluide en temps réel (15 min) sans recharger la page.
    \item \textbf{Scalabilité :} Supporter l'ingestion massive via Kafka et l'orchestration via Airflow.
    \item \textbf{Sécurité :} Accès protégé et réservé aux administrateurs pour l'infrastructure.
    \item \textbf{Ergonomie :} Interface intuitive et Responsive Design.
    \item \textbf{Traçabilité :} Isolation et journalisation des anomalies dans les flux de données.
\end{itemize}

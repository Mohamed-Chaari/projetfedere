# dags/src/analysis/peaks.py
def detect_peaks_fn():
    df = pd.read_sql("SELECT * FROM weather_historical", engine)
    df['z_score'] = (df['temperature'] - df['temperature'].mean()) / df['temperature'].std()
    peaks = df[df['z_score'] > 1.5]
    return {'total': len(peaks)}
\end{lstlisting}
